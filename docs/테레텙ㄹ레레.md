3. #### 코드

```cpp
#include <iostream>
#include <stack>

using namespace std;

stack<int> mono_stack;

int arr[6] = { 2, 7, 6, 3, 5, 9 };

int main() {
    for(int i = 0; i < 6; i++) {
        while(!mono_stack.empty() && mono_stack.top() > arr[i]){
          mono_stack.pop();
        }
        mono_stack.push(arr[i]);
    }
    while(!mono_stack.empty()){
      cout << mono_stack.top() << ' ';
      mono_stack.pop();
    }
    cout << '\n';
    return 0;
}
```
